# Firebase Configuration
# Copy this content to a file named .env in the root directory
# Get these values from: https://console.firebase.google.com → Project Settings → Your apps → Web app

VITE_FIREBASE_API_KEY=your_api_key_here

# Instructions:
# 1. Create a file named .env (with the dot at the start) in the root directory
